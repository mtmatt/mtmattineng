#include<bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f3f3f3f3f

long long tb[1005][1005],dp[1005][1005];

int main(){
ios::sync_with_stdio(0);cin.tie(0);

long long h,w,c;
cin>>h>>w>>c;
for(int i=1;i<=h;i++) for(int j=1;j<=w;j++){
    cin>>tb[i][j];
}
for(int i=0;i<=h;i++){
    dp[i][0]=INF;
}
for(int j=0;j<=w;j++){
    dp[0][j]=INF;
}

long long ans=INF;
for(int i=1;i<=h;i++){
    for(int j=1;j<=w;j++){
        dp[i][j]=min({tb[i][j],dp[i-1][j]+c,dp[i][j-1]+c});
        ans=min({ans,dp[i-1][j]+tb[i][j]+c,dp[i][j-1]+tb[i][j]+c});
    }
}

cout<<ans;
return 0;

}
